1. Introdução ao R

R é uma linguagem de programação e ambiente estatístico amplamente utilizado para análise de dados, visualização e modelagem estatística. É de código aberto e possui uma vasta comunidade de usuários.

2. Tipos de Dados

2.1 Tipos Básicos

# Numérico (números decimais)
x <- 10.5
class(x)  # "numeric"

# Inteiro
y <- 10L
class(y)  # "integer"

# Caractere (string)
z <- "Olá, Mundo!"
class(z)  # "character"

# Lógico (booleano)
w <- TRUE
class(w)  # "logical"
w2 <- FALSE

# Complexo
c <- 1 + 2i
class(c)  # "complex"

# Verificando tipos
is.numeric(x)  # TRUE
is.character(z)  # TRUE
is.logical(w)  # TRUE

3. Operadores Aritméticos

# Adição
5 + 3  # 8

# Subtração
10 - 4  # 6

# Multiplicação
7 * 2  # 14

# Divisão
15 / 3  # 5

# Potência
2 ^ 3  # 8
2 ** 3  # 8 (alternativa)

# Módulo (resto da divisão)
10 %% 3  # 1

# Divisão inteira
10 %/% 3  # 3

# Exemplos com variáveis
a <- 20
b <- 4
soma <- a + b
produto <- a * b

4. Operadores Lógicos

# Operadores de comparação
5 > 3   # TRUE
5 < 3   # FALSE
5 == 5  # TRUE (igualdade)
5 != 3  # TRUE (diferença)
5 >= 5  # TRUE
5 <= 4  # FALSE

# Operadores booleanos
TRUE & FALSE   # FALSE (E)
TRUE | FALSE   # TRUE (OU)
!TRUE          # FALSE (NÃO)
xor(TRUE, TRUE) # FALSE (OU exclusivo)

# Exemplos combinados
x <- 10
y <- 20
(x > 5) & (y < 30)  # TRUE
(x > 15) | (y == 20) # TRUE

6. Vetores

6.1 Criação de Vetores

# Vetor com função c() (combine)
numeros <- c(1, 2, 3, 4, 5)
letras <- c("a", "b", "c", "d")
logicos <- c(TRUE, FALSE, TRUE, TRUE)

# Vetores com sequências
1:10  # 1 até 10
seq(1, 10, by = 2)  # 1,3,5,7,9
seq(1, 10, length.out = 5)  # 1.00, 3.25, 5.50, 7.75, 10.00

# Vetores repetidos
rep(1:3, times = 2)  # 1,2,3,1,2,3
rep(1:3, each = 2)   # 1,1,2,2,3,3

6.2 Acesso aos Dados

vetor <- c(10, 20, 30, 40, 50)

# Acesso por índice (começa em 1)
vetor[1]      # 10 (primeiro elemento)
vetor[3]      # 30 (terceiro elemento)
vetor[c(1,3)] # 10, 30 (primeiro e terceiro)

# Acesso por condição
vetor[vetor > 25]  # 30, 40, 50

# Excluindo elementos
vetor[-1]     # 20, 30, 40, 50 (exceto primeiro)
vetor[-c(2,4)] # 10, 30, 50 (exceto segundo e quarto)

6.3 Operações com Vetores

v1 <- c(1, 2, 3)
v2 <- c(4, 5, 6)

# Operações aritméticas elemento a elemento
v1 + v2   # 5, 7, 9
v1 * 2    # 2, 4, 6
v1 ^ 2    # 1, 4, 9

# Funções úteis
length(v1)        # 3 (tamanho)
sum(v1)           # 6 (soma)
mean(v1)          # 2 (média)
sd(v1)            # desvio padrão
sort(v1)          # ordenação
rev(v1)           # reversão

6.4 Nomes em Vetores (cnames)

# Atribuindo nomes aos elementos
notas <- c(90, 85, 88)
names(notas) <- c("Matemática", "Português", "História")
notas
# Matemática  Português    História 
#        90         85         88

# Acesso por nome
notas["Matemática"]  # 90
notas[c("Português", "História")]  # 85, 88

# Criando vetor já com nomes
idades <- c(João = 25, Maria = 30, Pedro = 35)

7. Matrizes

7.1 Criação de Matrizes

# Matriz a partir de vetor
m <- matrix(1:12, nrow = 3, ncol = 4)
m
#      [,1] [,2] [,3] [,4]
# [1,]    1    4    7   10
# [2,]    2    5    8   11
# [3,]    3    6    9   12

# Por linhas (byrow = TRUE)
m2 <- matrix(1:12, nrow = 3, ncol = 4, byrow = TRUE)
m2

# Combinando vetores
v1 <- 1:3
v2 <- 4:6
rbind(v1, v2)  # por linhas
cbind(v1, v2)  # por colunas

7.2 Acesso e Operações

m <- matrix(1:9, nrow = 3, ncol = 3)

# Acesso a elementos
m[2, 3]      # elemento linha 2, coluna 3
m[2, ]       # toda linha 2
m[, 3]       # toda coluna 3
m[1:2, 2:3]  # submatriz

# Dimensões
dim(m)       # 3, 3
nrow(m)      # 3
ncol(m)      # 3

# Operações
m * 2        # multiplicação escalar
t(m)         # transposta
diag(m)      # diagonal principal

# Nomes em matrizes
rownames(m) <- c("Linha1", "Linha2", "Linha3")
colnames(m) <- c("Col1", "Col2", "Col3")
m["Linha1", "Col2"]  # acesso por nome

8. Listas

8.1 Criação e Acesso

# Lista heterogênea
lista <- list(
  nome = "Maria",
  idade = 30,
  notas = c(7.5, 8.0, 9.2),
  aprovado = TRUE
)

# Acesso
lista$nome           # "Maria"
lista[["idade"]]     # 30
lista[[3]]           # vetor de notas
lista[[3]][2]        # 8.0 (segunda nota)

# Estrutura da lista
str(lista)
length(lista)        # 4 elementos

8.2 Operações com Listas

# Adicionando elementos
lista$curso <- "Estatística"

# Removendo elementos
lista$aprovado <- NULL

# Lista de listas
lista2 <- list(
  pessoa1 = list(nome = "João", idade = 25),
  pessoa2 = list(nome = "Ana", idade = 28)
)

# Acesso em listas aninhadas
lista2$pessoa1$nome  # "João"

9. Strings (Caracteres)

9.1 Manipulação Básica

# Criação
s1 <- "Hello"
s2 <- 'World'
s3 <- paste(s1, s2)  # "Hello World"
s4 <- paste0(s1, s2) # "HelloWorld"

# Concatenação com separador
paste(c("A", "B", "C"), collapse = ", ")  # "A, B, C"

# Número de caracteres
nchar("R Programming")  # 13

# Maiúsculas/minúsculas
toupper("hello")  # "HELLO"
tolower("HELLO")  # "hello"

9.2 Extração e Substituição

texto <- "Programação em R é divertido"

# Substring
substr(texto, 1, 11)  # "Programação"

# Substituição
sub("divertido", "poderoso", texto)  # substitui primeira ocorrência
gsub(" ", "_", texto)  # substitui todas as ocorrências

# Divisão
strsplit(texto, " ")  # lista com palavras

# Busca
grepl("R", texto)     # TRUE (contém "R")
grep("R", c("Python", "R", "Java"))  # 2 (posição)

10. Data Frames

10.1 Criação e Estrutura

# Criando data frame
df <- data.frame(
  nome = c("Ana", "Bruno", "Carla", "Daniel"),
  idade = c(25, 30, 35, 28),
  altura = c(1.65, 1.80, 1.70, 1.75),
  stringsAsFactors = FALSE
)

# Visualização
View(df)        # abre visualizador
head(df)        # primeiras linhas
tail(df)        # últimas linhas
str(df)         # estrutura
summary(df)     # resumo estatístico

# Dimensões
dim(df)         # 4 linhas, 3 colunas
nrow(df)        # 4
ncol(df)        # 3
names(df)       # nomes das colunas

10.2 Acesso e Manipulação

# Acesso por coluna
df$nome          # vetor com nomes
df[["idade"]]    # vetor com idades
df[, "altura"]   # vetor com alturas

# Acesso por linha/coluna
df[2, ]           # segunda linha
df[, 2]           # segunda coluna
df[3, 2]          # terceira linha, segunda coluna (idade da Carla)

# Subconjuntos
subset(df, idade > 28)  # pessoas com mais de 28 anos
df[df$altura > 1.70, ]  # pessoas com mais de 1.70m

# Adicionando colunas
df$imc <- c(22, 25, 23, 24)  # nova coluna

# Ordenação
df[order(df$idade), ]        # ordena por idade crescente
df[order(-df$idade), ]       # ordena por idade decrescente

10.3 Operações com Data Frames

# Estatísticas básicas
mean(df$idade)        # média das idades
sd(df$altura)         # desvio padrão das alturas

# Adicionando linhas
nova_linha <- data.frame(nome = "Elena", idade = 32, altura = 1.68, imc = 22.5)
df <- rbind(df, nova_linha)

# Combinando data frames
df2 <- data.frame(
  nome = c("Fernando", "Gabriela"),
  idade = c(40, 27),
  altura = c(1.82, 1.63),
  imc = c(26, 21)
)
df_completo <- rbind(df, df2)

# Exportação/Importação
write.csv(df, "dados.csv", row.names = FALSE)
df_lido <- read.csv("dados.csv")
